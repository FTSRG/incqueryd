\newcommand{\picBig}[2]{

		\centering
		\begin{adjustbox}{max size={\textwidth}{\textheight}}
			\includegraphics[scale=1.3]{figures/#1}
		\end{adjustbox}

		\label{fig:#1}
	\end{figure}}
